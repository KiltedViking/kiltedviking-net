Case-satser i JavaScript

Kod:

<script type="text/javascript">
  var min_var = 3; //deklarera variabel
  var min_text;

  switch(min_var*2) //Testa i case-sats om ...
  {
    case 4: //...talet är 4
      min_text = "Villkoret (" + min_var + "*2) är 4";
      break;
    case 6: //...talet är 6
      min_text = "Villkoret (" + min_var + "*2) är 6";
      break;
    case 8: //...talet är 8
      min_text = "Villkoret (" + min_var + "*2) är 8";
      break;
    default: //...annars skriv ut följande
      min_text = "Alternativet fanns inte i case-satsen";
  }

  document.writeln(min_text); //Skriv ut resultatet till hemsida
</script>
