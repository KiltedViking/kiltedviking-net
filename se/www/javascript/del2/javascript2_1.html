If-satser i JavaScript

Version 1 - "fullständig" "större" if-sats

Kod:
<script>
  var min_var = 4;   //deklarera variabler
  var min_text;

  if(min_var * 2 == 6) { //Testa i if-sats om sant...
    min_text = "Villkoret (" + min_var + " * 2 == 6) var sant";
  }
  else {                 //...annars är det falskt
    min_text = "Villkoret (" + min_var + " * 2 == 6) var falskt";
  }

  document.writeln(min_text);  //Skriv ut resultatet till hemsida
</script>

Version 3 - "mindre" version av if-sats

Kod:
<script>
  var min_var = 3;   //deklarera variabler
  var min_text1, min_text2;

  min_text1 = "Villkoret (" + min_var + " * 2 == 6) var ";
  min_text2 = (min_var * 2 == 6) ? "sant" : "falskt";

  document.writeln(min_text1 + min_text2);  //Skriv ut resultatet till hemsida
</script>
